\label{sec:data}

The data package collects, aggregates, and persists simulation metrics.

% ─────────────────────────────────────────────────────────────────────────────
\subsection{\texttt{GenerationMetrics} (struct)}

Plain value type returned by \texttt{MetricsCollector::collect()}.

\begin{table}[H]
\centering\small
\caption{\texttt{GenerationMetrics} fields}
\begin{tabularx}{\textwidth}{@{}lX@{}}
\toprule
\textbf{Field} & \textbf{Description} \\
\midrule
\texttt{generation}          & Generation index \\
\texttt{predator\_count}     & Alive predators at end of generation \\
\texttt{prey\_count}         & Alive prey at end of generation \\
\texttt{best\_fitness}       & Maximum genome fitness in population \\
\texttt{avg\_fitness}        & Mean genome fitness in population \\
\texttt{num\_species}        & Active species count \\
\texttt{avg\_genome\_complexity} & Mean \texttt{Genome::complexity()} across population \\
\bottomrule
\end{tabularx}
\end{table}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{\texttt{MetricsCollector} (class)}

\begin{table}[H]
\centering\small
\caption{\texttt{MetricsCollector} interface}
\begin{tabularx}{\textwidth}{@{}lX@{}}
\toprule
\textbf{Method} & \textbf{Description} \\
\midrule
\texttt{collect(generation, population, pred\_count, prey\_count)} &
  Iterate \texttt{population} to find max/avg fitness and avg complexity;
  append to \texttt{history\_}; return metrics struct. \\
\texttt{history()} &
  All metrics collected since construction. \\
\bottomrule
\end{tabularx}
\end{table}

\textbf{Private:} \texttt{history\_: vector<GenerationMetrics>}.

% ─────────────────────────────────────────────────────────────────────────────
\subsection{\texttt{Logger} (class)}

Writes all run output to \texttt{output/[YYYYMMDD\_HHMMSS\_seedN]/}.

\begin{table}[H]
\centering\small
\caption{\texttt{Logger} interface}
\begin{tabularx}{\textwidth}{@{}lX@{}}
\toprule
\textbf{Method} & \textbf{Description} \\
\midrule
\texttt{Logger(output\_dir, seed)} & Store base dir and seed; do not open files yet \\
\texttt{initialize(config)} & Create run directory; open all files; write headers; save \texttt{config.json} \\
\texttt{log\_generation(...)} & Append one row to \texttt{stats.csv} \\
\texttt{log\_best\_genome(generation, genome)} & Append JSON entry to \texttt{genomes.json} \\
\texttt{log\_species(generation, species)} & Append rows to \texttt{species.csv} \\
\texttt{log\_tick(generation, tick, agents)} & Append rows to \texttt{ticks\_buffer\_}; auto-flush every 500 rows \\
\texttt{flush\_ticks()} & Write \texttt{ticks\_buffer\_} to \texttt{ticks.csv} and reset buffer \\
\texttt{flush()} & Flush all open file streams \\
\texttt{run\_dir()} & Path to the created run directory \\
\bottomrule
\end{tabularx}
\end{table}

\textbf{Private members:}
\texttt{base\_dir\_}, \texttt{run\_dir\_}: \texttt{string};
\texttt{seed\_}: \texttt{uint64\_t};
\texttt{stats\_file\_}, \texttt{genomes\_file\_}, \texttt{species\_file\_}, \texttt{ticks\_file\_}:
\texttt{std::ofstream};
\texttt{ticks\_buffer\_}: \texttt{string} (accumulated CSV rows);
\texttt{ticks\_buffered\_}: \texttt{int};
\texttt{genomes\_first\_entry\_}: \texttt{bool} (for JSON comma handling);
\texttt{TICK\_FLUSH\_EVERY = 500}: \texttt{constexpr int}.

\subsubsection{Output File Formats}

\paragraph{\texttt{config.json}} Full \texttt{SimulationConfig} snapshot, written by
\texttt{Logger::initialize()}.

\paragraph{\texttt{stats.csv}}
\begin{lstlisting}[language={}]
generation,predator_count,prey_count,best_fitness,avg_fitness,num_species,avg_complexity
0,50,150,12.34,5.67,8,16.2
...
\end{lstlisting}

\paragraph{\texttt{species.csv}}
\begin{lstlisting}[language={}]
generation,species_id,size,avg_fitness,stagnation_count
0,1,42,8.1,0
...
\end{lstlisting}

\paragraph{\texttt{genomes.json}} JSON array of best-genome snapshots:
\begin{lstlisting}[language={}]
[
  {"generation": 0, "genome": {
    "nodes": [{"id": 0, "type": "input"}, ...],
    "connections": [{"in": 0, "out": 16, "weight": 0.42,
                     "enabled": true, "innovation": 3}, ...]
  }},
  ...
]
\end{lstlisting}

\paragraph{\texttt{ticks.csv} (optional)} Written when \texttt{tick\_log\_enabled: true}:
\begin{lstlisting}[language={}]
generation,tick,agent_id,type,alive,x,y,energy,kills,food_eaten
0,0,0,predator,1,312.5,201.3,100.0,0,0
...
\end{lstlisting}
Rows are buffered in \texttt{ticks\_buffer\_} (string) and flushed every
\texttt{TICK\_FLUSH\_EVERY = 500} rows to amortize \texttt{write()} syscall cost.
